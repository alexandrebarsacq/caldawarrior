**Test 1: `test_parse_multistatus_large`** (COMPAT-01: large responses)

Build an XML string with 25 `<D:response>` elements, each containing a valid VTODO with a unique UID (`uid-001` through `uid-025`). Use a loop to generate:

```rust
#[test]
fn test_parse_multistatus_large() {
    let mut xml = String::from(
        r#"<?xml version="1.0"?><D:multistatus xmlns:D="DAV:" xmlns:C="urn:ietf:params:xml:ns:caldav">"#
    );
    for i in 1..=25 {
        xml.push_str(&format!(
            r#"<D:response><D:href>/cal/uid-{i:03}.ics</D:href><D:propstat><D:prop><D:getetag>"etag-{i:03}"</D:getetag><C:calendar-data>BEGIN:VCALENDAR
VERSION:2.0
BEGIN:VTODO
UID:uid-{i:03}
SUMMARY:Task {i}
END:VTODO
END:VCALENDAR</C:calendar-data></D:prop></D:propstat></D:response>"#
        ));
    }
    xml.push_str("</D:multistatus>");

    let results = parse_multistatus_vtodos(&xml);
    assert_eq!(results.len(), 25, "should parse all 25 VTODOs");
    assert_eq!(results[0].vtodo.uid, "uid-001");
    assert_eq!(results[24].vtodo.uid, "uid-025");
    // Verify no UIDs are missing
    for (i, fv) in results.iter().enumerate() {
        assert_eq!(fv.vtodo.uid, format!("uid-{:03}", i + 1));
    }
}
```

#[test]
fn test_parse_multistatus_special_chars() {
    let xml = r#"<?xml version="1.0"?>
<D:multistatus xmlns:D="DAV:" xmlns:C="urn:ietf:params:xml:ns:caldav">
  <D:response>
    <D:href>/cal/special-uid.ics</D:href>
    <D:propstat>
      <D:prop>
        <D:getetag>"special-etag"</D:getetag>
        <C:calendar-data>BEGIN:VCALENDAR
VERSION:2.0
BEGIN:VTODO
UID:special-uid
SUMMARY:Buy eggs &amp; milk &lt;urgent&gt;
DESCRIPTION:Caf\u00e9 meeting \u2014 don\u2019t forget!
END:VTODO
END:VCALENDAR</C:calendar-data>
      </D:prop>
    </D:propstat>
  </D:response>
</D:multistatus>"#;

    let results = parse_multistatus_vtodos(xml);
    assert_eq!(results.len(), 1, "should parse VTODO with special characters");
    assert_eq!(results[0].vtodo.uid, "special-uid");
    // quick-xml decodes &amp; -> & and &lt; -> < and &gt; -> >
    // The iCal content after XML decode goes to from_icalendar_string
    let summary = results[0].vtodo.summary.as_deref().unwrap_or("");
    assert!(summary.contains("&"), "ampersand should be decoded from &amp;");
    assert!(summary.contains("<"), "angle bracket should be decoded from &lt;");
}
```

Note: The XML `&amp;` is decoded by quick-xml to `&` in the calendar-data text. Then `from_icalendar_string` receives the decoded iCal text.

def put_vtodo_raw_ical(self, collection_url, uid, ical_text):
    """PUT raw iCalendar text directly, for compatibility edge-case tests.

    This keyword enables tests with exact iCal content that cannot be
    produced by the structured put_vtodo_with_fields keyword, such as
    VALUE=DATE parameters, TZID parameters, and arbitrary X-properties.

    Args:
        collection_url: Full URL of the CalDAV collection.
        uid: Unique identifier for the VTODO (used as the filename).
        ical_text: Complete iCalendar text to PUT.

    Raises:
        AssertionError: On non-2xx HTTP response.
    """
    url = f"{collection_url}{uid}.ics"
    response = self._session.put(
        url,
        data=ical_text.encode('utf-8'),
        headers={'Content-Type': 'text/calendar; charset=utf-8'},
    )
    self._check_response(response)
```

**Important implementation notes:**

```robot
${ical} =    Evaluate    "BEGIN:VCALENDAR\\r\\nVERSION:2.0\\r\\n..."
```

OR, if the `put_vtodo_raw_ical` keyword works with just `\n` (Radicale accepts both), use the simpler SEPARATOR approach:

```robot
${ical} =    Catenate    SEPARATOR=\n
...    BEGIN:VCALENDAR
...    VERSION:2.0
...    PRODID:-//test//EN
...    BEGIN:VTODO
...    UID:date-only-due-001
...    SUMMARY:Date only task
...    DUE;VALUE=DATE:20260315
...    STATUS:NEEDS-ACTION
...    DTSTAMP:20260319T100000Z
...    END:VTODO
...    END:VCALENDAR
```
